	\nameref{herb:#1}%

	\nameref{herbbare:#1}%
}

\makeatletter
\newcommand\herb[3][\@nil]{%A herb, with identity and type
	\def\akaname{#1}%Optional argument is an alternative name for the herb.
	#2 (%
	\ifx\akaname\@nnil%
		%
	\else%
		a.k.a.\ \akaname,\ %
	\fi%
	\herbtype{#3})%
}
\newcommand\herbs[2]{%
	#1 (\herbtypeplural{#2})%
}

}
